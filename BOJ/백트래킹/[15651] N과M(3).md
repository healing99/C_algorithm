#include <iostream>
using namespace std;

int n, m;
int res[10];

void DFS(int L) {
	if (L == m) {
		for (int i = 0; i < m; i++) {
			cout << res[i] << " ";
		}
		puts("");
	}
	else {
		for (int i = 1; i <= n; i++) {
			res[L] = i;
			DFS(L + 1);
		}
	}
}

int main() {
	cin >> n >> m;
	DFS(0);
}
